18  	public void flightDetails(TripType type, String passenger)
19  			throws InterruptedException, InvalidRadioButtonException {
20  
21  		switch (type) {
22  		case RoundTrip:
23  			driver.findElement(By.xpath("//input[@value='roundtrip']")).click();
24  			break;
25  		case OneWay:
26  			driver.findElement(By.xpath("//input[@value='oneway']")).click();
27  			break;
28  		default:
29  			throw new InvalidRadioButtonException();
30  		}
31  
32  		WebElement selectPassenger = driver.findElement(By.xpath("//select[@name='passCount']"));
33  		Select dropDown1 = new Select(selectPassenger);
34  		dropDown1.selectByValue(passenger);
35  		Thread.sleep(3000);
36  	}

81  			driver.findElement(By.xpath("//input[@value='Business']")).click();
