16  	public void passengerData(String firstName, String lastName, String meal) {
17  		driver.findElement(By.xpath("//input[@name='passFirst0']")).sendKeys(firstName);
18  		driver.findElement(By.xpath("//input[@name='passLast0']")).sendKeys(lastName);
19  
20  		WebElement selectMeal = driver.findElement(By.xpath("//select[@name='pass.0.meal']"));
21  		Select dropdown = new Select(selectMeal);
22  		dropdown.selectByVisibleText(meal);
23  	}

25  	public void creditCardData(String creditType, String number, String expirationDate, String expirationYear) {
26  		WebElement creditCardType = driver.findElement(By.xpath("//select[@name='creditCard']"));
27  		Select dropdown1 = new Select(creditCardType);
28  		dropdown1.selectByVisibleText(creditType);
29  
30  		driver.findElement(By.xpath("//input[@name='creditnumber']")).sendKeys(number);
31  
32  		WebElement creditCardExpirationDate = driver.findElement(By.xpath("//select[@name='cc_exp_dt_mn']"));
33  		Select dropdown2 = new Select(creditCardExpirationDate);
34  		dropdown2.selectByValue(expirationDate);
35  
36  		WebElement creditCardExpirationYear = driver.findElement(By.xpath("//select[@name='cc_exp_dt_yr']"));
37  		Select dropdown3 = new Select(creditCardExpirationYear);
38  		dropdown3.selectByValue(expirationYear);
39  	}

47  	public void ticketLessTravel(boolean checked) {
48  		WebElement ticketLessTravel = driver
49  				.findElement(By.xpath("//font[contains(text(),'Ticketless')]/preceding-sibling::input"));
50  		if (checked) {
51  			if (!ticketLessTravel.isSelected()) {
52  				ticketLessTravel.click();
53  			}
54  		} else {
55  			if (ticketLessTravel.isSelected()) {
56  				ticketLessTravel.click();
57  			}
58  		}
59  	}

61  	public void billingAddress(String address, String city, String postalCode) {

68  	}
